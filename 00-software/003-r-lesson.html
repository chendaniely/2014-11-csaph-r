1.2.1 50

# write a function, that takes in a number, and returns TRUE
# if it is less than 50
# returns FALSE if >= 50
cut_50 <- function(number){
  if(number < 50){
    print(TRUE)
  } else{
    print(FALSE)
  }
}
cut_50(0)
## [1] TRUE
cut_50(50)
## [1] FALSE
cut_50(100)
## [1] FALSE

1.2.2 50 with neg

# write a function that takes in a number, returns "neg"
# if < 0
# returns "SMALL" if >= 0 and < 50
# returns "big" if >= 50
cut_50_neg <- function(number){
  if(number < 0){
    print("neg")
  } else if (number < 50){
    print("SMALL")
  } else {
    print("big")
  }
}

cut_50_neg(-1)
## [1] "neg"
cut_50_neg(0)
## [1] "SMALL"
cut_50_neg(50)
## [1] "big"
